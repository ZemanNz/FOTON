#include <Arduino.h>
#include <Wire.h>
#include <Adafruit_MPU6050.h>
#include <Adafruit_TCS34725.h>
#include <Adafruit_Sensor.h>

// Definice pinů pro I2C sběrnici (jediná používaná sběrnice)
#define I2C_SDA 21
#define I2C_SCL 22

// Globální instance senzorů na jedné sběrnici
Adafruit_MPU6050 mpu;
Adafruit_TCS34725 tcs = Adafruit_TCS34725(TCS34725_INTEGRATIONTIME_50MS, TCS34725_GAIN_1X);

// Funkce pro skenování I2C adres na sběrnici
void scanI2CBus(TwoWire &bus, const char* name) {
    Serial.printf("--- Skenuji I2C sbernici: %s ---\n", name);
    int count = 0;
    for (uint8_t address = 1; address < 127; address++) {
        bus.beginTransmission(address);
        uint8_t error = bus.endTransmission();
        if (error == 0) {
            Serial.printf("  Nalezeno zarizeni na adrese 0x%02X\n", address);
            count++;
        } else if (error == 4) {
            Serial.printf("  Neznama chyba na adrese 0x%02X\n", address);
        }
    }
    if (count == 0) {
        Serial.println("  Zadna zarizeni nenalezena.");
    }
    Serial.println("-------------------------------------");
}

void setup() {
    Serial.begin(115200);
    delay(1000);
    Serial.println("=== Bare ESP32 Test: MPU6050 & TCS34725 na jedne sbernici ===");

    // Nastavení pinů pro I2C sběrnici na INPUT_PULLUP
    pinMode(I2C_SDA, INPUT_PULLUP);
    pinMode(I2C_SCL, INPUT_PULLUP);
    delay(50); // Stabilizace pinů

    // Inicializace I2C sběrnice
    Wire.begin(I2C_SDA, I2C_SCL, 100000);

    // Skenování sběrnice před inicializací čipů
    scanI2CBus(Wire, "Wire (Piny 14/26)");

    // 1. Inicializace MPU6050
    Serial.println("Inicializuji MPU6050...");
    bool mpu_status = false;
    if (mpu.begin(0x68, &Wire)) {
        mpu_status = true;
    } else if (mpu.begin(0x69, &Wire)) {
        mpu_status = true;
    }

    // Adafruit knihovny mohou resetovat I2C piny na výchozí (21/22), musíme je znovu nastavit

    if (mpu_status) {
        Serial.println("MPU6050 inicializovan uspesne.");
        mpu.setAccelerometerRange(MPU6050_RANGE_2_G);
        mpu.setGyroRange(MPU6050_RANGE_250_DEG);
        mpu.setFilterBandwidth(MPU6050_BAND_21_HZ);
    } else {
        Serial.println("CHYBA: MPU6050 nenalezen! Zkontrolujte zapojeni.");
    }

    // 2. Inicializace TCS34725
    Serial.println("Inicializuji TCS34725...");
    bool tcs_status = tcs.begin(TCS34725_ADDRESS, &Wire);
    
    // Znovu nastavíme piny pro Wire, pokud je knihovna přepsala

    if (tcs_status) {
        Serial.println("TCS34725 inicializovan uspesne.");
    } else {
        Serial.println("CHYBA: TCS34725 nenalezen!");
    }

    Serial.println("=== Inicializace dokoncena. Spoustim hlavni smycku ===");
}

void loop() {
    static unsigned long last_print = 0;
    
    // Čtení a výpis dat každých 500 ms
    if (millis() - last_print >= 500) {
        last_print = millis();
        Serial.println("\n--- DATA ZE SENZORU ---");

        // 1. Ctení MPU6050
        sensors_event_t a, g, temp;
        if (mpu.getEvent(&a, &g, &temp)) {
            Serial.printf("MPU6050: Accel X: %.2f, Y: %.2f, Z: %.2f m/s^2 | Gyro X: %.2f, Y: %.2f, Z: %.2f rad/s | Temp: %.1f C\n",
                          a.acceleration.x, a.acceleration.y, a.acceleration.z,
                          g.gyro.x, g.gyro.y, g.gyro.z,
                          temp.temperature);
        } else {
            Serial.println("MPU6050: Chyba cteni dat!");
        }

        // 2. Ctení TCS34725
        uint16_t r, g_val, b, c;
        tcs.getRawData(&r, &g_val, &b, &c);
        float r_p = 0, g_p = 0, b_p = 0;
        tcs.getRGB(&r_p, &g_p, &b_p);
        Serial.printf("TCS34725:  R: %d, G: %d, B: %d, C: %d | Procenta -> R: %.1f, G: %.1f, B: %.1f\n",
                      r, g_val, b, c, r_p, g_p, b_p);
    }
    delay(10);
}
